// ===== ListNode CLASS (given in LeetCode, know it by heart!) =====
class ListNode {
    int val;
    ListNode next;
    ListNode() {}
    ListNode(int val) { this.val = val; }
    ListNode(int val, ListNode next) { this.val = val; this.next = next; }
}

// ===== CREATE A LINKED LIST =====
ListNode head = new ListNode(1);
head.next     = new ListNode(2);
head.next.next = new ListNode(3); // 1 -> 2 -> 3 -> null

// ===== TRAVERSE =====
ListNode curr = head;
while (curr != null) {
    System.out.println(curr.val);
    curr = curr.next;
}

// ===== FIND LENGTH =====
int len = 0;
ListNode curr = head;
while (curr != null) { len++; curr = curr.next; }

// ===== FIND MIDDLE (LC #876) — Fast & Slow pointers =====
ListNode slow = head, fast = head;
while (fast != null && fast.next != null) {
    slow = slow.next;
    fast = fast.next.next;
}
// slow is now at middle (for even length: second middle)

// ===== REVERSE LINKED LIST (LC #206) =====
ListNode prev = null, curr = head;
while (curr != null) {
    ListNode next = curr.next;
    curr.next = prev;
    prev = curr;
    curr = next;
}
return prev; // new head

// ===== DETECT CYCLE (LC #141) =====
ListNode slow = head, fast = head;
while (fast != null && fast.next != null) {
    slow = slow.next; fast = fast.next.next;
    if (slow == fast) return true;
}
return false;

// ===== ADD DUMMY HEAD (essential pattern!) =====
// Simplifies edge cases when result might be at head
ListNode dummy = new ListNode(0);
dummy.next = head;
ListNode curr = dummy;
// ... do work ...
return dummy.next; // actual head of result

    // Remove elements outside window
    while (!dq.isEmpty() && dq.peekFirst() < i - k + 1)
        dq.pollFirst();
    // Remove smaller elements from back (maintain decreasing order)
    while (!dq.isEmpty() && nums[dq.peekLast()] < nums[i])

    dq.offerLast(i);
    // Window is full
    if (i >= k - 1) result[i - k + 1] = nums[dq.peekFirst()];
}

class TrieNode {
    TrieNode[] children = new TrieNode[26]; // for lowercase a-z
    boolean isEnd = false;                  // marks end of word
}

class Trie {
    private TrieNode root;

    public Trie() {
        root = new TrieNode();
    }

    // Insert word - O(L)
    public void insert(String word) {
        TrieNode node = root;
        for (char c : word.toCharArray()) {
            int idx = c - 'a';
            if (node.children[idx] == null)
                node.children[idx] = new TrieNode();
            node = node.children[idx];
        }
        node.isEnd = true;
    }

    // Search exact word - O(L)
    public boolean search(String word) {
        TrieNode node = root;
        for (char c : word.toCharArray()) {
            int idx = c - 'a';
            if (node.children[idx] == null) return false;
            node = node.children[idx];
        }
        return node.isEnd; // must be marked as end of word
    }

    // Search prefix (startsWith) - O(L)
    public boolean startsWith(String prefix) {
        TrieNode node = root;
        for (char c : prefix.toCharArray()) {
            int idx = c - 'a';
            if (node.children[idx] == null) return false;
            node = node.children[idx];
        }
        return true; // prefix exists (don't need isEnd)
    }

    // Helper: traverse to end of prefix, return node
    private TrieNode find(String prefix) {
        TrieNode node = root;
        for (char c : prefix.toCharArray()) {
            int idx = c - 'a';
            if (node.children[idx] == null) return null;
            node = node.children[idx];
        }
        return node;
    }

    // Count words with given prefix
    public int countWordsWithPrefix(String prefix) {
        TrieNode node = find(prefix);
        if (node == null) return 0;
        return countWords(node);
    }

    private int countWords(TrieNode node) {
        int count = node.isEnd ? 1 : 0;
        for (TrieNode child : node.children)
            if (child != null) count += countWords(child);
        return count;
    }
}

Union-Find (Disjoint Set)
חיבור רכיבים • find/union כמעט O(1)
DS▼
מה זה? מבנה נתונים לניהול קבוצות מנותקות (Disjoint Sets). תומך בשתי פעולות: find (מצא את שורש הקבוצה) ו-union (חבר שתי קבוצות). עם path compression ו-union by rank, כל פעולה כמעט O(1). מתי? Connected components, detect cycles, Kruskal's MST, accounts merge, number of islands (חלופה ל-BFS/DFS).
☕ מימוש מלא עם Path Compression + Union by Rank
class UnionFind {
    private int[] parent, rank;
    private int components; // number of connected components

    public UnionFind(int n) {
        parent = new int[n];
        rank   = new int[n];
        components = n;
        for (int i = 0; i < n; i++) parent[i] = i; // each node is its own parent
    }

    // Find root with PATH COMPRESSION (flattens tree)
    public int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]); // recursive compression
        return parent[x];
    }

    // Union by RANK (attach smaller tree under larger tree)
    public boolean union(int x, int y) {
        int rootX = find(x), rootY = find(y);
        if (rootX == rootY) return false; // already connected!
        if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
        components--;
        return true; // successfully merged
    }

    // Check if x and y are in same component
    public boolean isConnected(int x, int y) {
        return find(x) == find(y);
    }

    // Get number of connected components
    public int getComponents() {
        return components;
    }
}

uf.union(0, 1);       // connect node 0 and 1
uf.union(1, 2);       // connect node 1 and 2
uf.isConnected(0, 2); // true (0,1,2 all connected)
uf.getComponents();   // number of separate groups

        if (grid[r][c] == '1') {
            int id = r * cols + c; // flatten 2D to 1D
            if (r+1 < rows && grid[r+1][c] == '1') uf.union(id, (r+1)*cols+c);
            if (c+1 < cols && grid[r][c+1] == '1') uf.union(id, r*cols+c+1);
        }
    }
}
// count components with '1' cells
// (need to count unique roots among '1' cells)

// ===== DETECT CYCLE in undirected graph =====
// union returns false if nodes already connected -> cycle!
for (int[] edge : edges) {
    if (!uf.union(edge[0], edge[1])) return true; // cycle detected
}
Path Compression + Union by Rank יחד = O(α(n)) amortized, כלומר כמעט O(1). α = inverse Ackermann function — מעשית קבוע <= 5 לכל n מעשי. זהו מבנה נתונים מרשים מאוד שפותר קונקטיביות ב-O(1)!

QuickSort
Divide & Conquer • O(n log n) avg • O(n²) worst
Sort▼
רעיון: בחר pivot, חלק מערך ל"קטנים מ-pivot" ו"גדולים מ-pivot", מיין רקורסיבית. In-place (O(log n) stack space). Worst case O(n²) כשה-pivot גרוע (מסודר כבר). מתי ללמד? ראיונות, QuickSelect (Kth element O(n) avg).
void quickSort(int[] arr, int left, int right) {
    if (left >= right) return;
    int pivotIdx = partition(arr, left, right);
    quickSort(arr, left, pivotIdx - 1);
    quickSort(arr, pivotIdx + 1, right);
}

int partition(int[] arr, int left, int right) {
    int pivot = arr[right]; // choose last as pivot (simple)
    int i = left - 1;       // pointer to smaller elements
    for (int j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            i++;
            int tmp = arr[i]; arr[i] = arr[j]; arr[j] = tmp;
        }
    }
    int tmp = arr[i+1]; arr[i+1] = arr[right]; arr[right] = tmp;
    return i + 1; // pivot final index
}

// ===== QuickSelect — Kth largest element O(n) avg (LC #215) =====
int quickSelect(int[] nums, int left, int right, int k) {
    int pivot = nums[right], i = left - 1;
    for (int j = left; j < right; j++) {
        if (nums[j] >= pivot) { // descending (find kth largest)
            i++;
            int tmp = nums[i]; nums[i] = nums[j]; nums[j] = tmp;
        }
    }
    i++;
    int tmp = nums[i]; nums[i] = nums[right]; nums[right] = tmp;
    if (i == k - 1) return nums[i];
    if (i > k - 1) return quickSelect(nums, left, i - 1, k);
    return quickSelect(nums, i + 1, right, k);
}

void mergeSort(int[] arr, int left, int right) {
    if (left >= right) return;
    int mid = left + (right - left) / 2;
    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}

void merge(int[] arr, int left, int mid, int right) {
    int[] temp = new int[right - left + 1];
    int i = left, j = mid + 1, k = 0;
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) temp[k++] = arr[i++];
        else                   temp[k++] = arr[j++];
    }
    while (i <= mid)   temp[k++] = arr[i++];
    while (j <= right) temp[k++] = arr[j++];
    for (int x = 0; x < temp.length; x++) arr[left + x] = temp[x];
}

// ===== Count Inversions (pairs where i<j but arr[i]>arr[j]) =====
int countInversions(int[] arr, int left, int right) {
    if (left >= right) return 0;
    int mid = left + (right - left) / 2;
    int count = countInversions(arr, left, mid)
              + countInversions(arr, mid + 1, right);
    // count during merge: every time we pick from right, all remaining left elements are inversions
    // (standard merge with counter)
    return count + mergeCount(arr, left, mid, right);
}

// Manual in-place HeapSort (O(1) space):
void heapSort(int[] arr) {
    int n = arr.length;
    // Build max-heap
    for (int i = n/2 - 1; i >= 0; i--) heapify(arr, n, i);
    // Extract elements from heap one by one
    for (int i = n - 1; i > 0; i--) {
        int tmp = arr[0]; arr[0] = arr[i]; arr[i] = tmp; // swap root to end
        heapify(arr, i, 0); // heapify reduced heap
    }
}

void heapify(int[] arr, int n, int i) {
    int largest = i, left = 2*i+1, right = 2*i+2;
    if (left < n && arr[left] > arr[largest]) largest = left;
    if (right < n && arr[right] > arr[largest]) largest = right;
    if (largest != i) {
        int tmp = arr[i]; arr[i] = arr[largest]; arr[largest] = tmp;
        heapify(arr, n, largest);
    }
}

// ===== Top K Frequent — Bucket Sort by Frequency (LC #347) =====
// Bucket index = frequency, O(n) time!
int[] topKFrequent(int[] nums, int k) {
    Map<Integer, Integer> freq = new HashMap<>();
    for (int n : nums) freq.put(n, freq.getOrDefault(n, 0) + 1);

    // Buckets: index = frequency (0 to nums.length)
    List<Integer>[] buckets = new List[nums.length + 1];
    for (int i = 0; i <= nums.length; i++) buckets[i] = new ArrayList<>();
    for (int num : freq.keySet()) buckets[freq.get(num)].add(num);

    int[] result = new int[k];
    int idx = 0;
    for (int f = buckets.length - 1; f > 0 && idx < k; f--)
        for (int num : buckets[f])
            if (idx < k) result[idx++] = num;
    return result;
}

// ===== General Bucket Sort for floats [0,1) =====
void bucketSort(float[] arr) {
    int n = arr.length;
    List<Float>[] buckets = new List[n];
    for (int i = 0; i < n; i++) buckets[i] = new ArrayList<>();
    for (float v : arr) buckets[(int)(v * n)].add(v);
    int idx = 0;
    for (List<Float> b : buckets) { Collections.sort(b); for (float v : b) arr[idx++] = v; }
}

~a       // NOT   — flip all bits
a << n   // Left shift  — multiply by 2^n
a >> n   // Right shift (signed) — divide by 2^n (keeps sign)
a >>> n  // Right shift (unsigned) — fill with 0 (even for negative)

// ===== COMMON BIT TRICKS =====
n & 1                  // check if odd (1) or even (0)
n & (n-1)              // clear lowest set bit
n & (-n)               // isolate lowest set bit
(n & (n-1)) == 0       // is power of 2? (and n>0)
n >> 1                 // divide by 2 (integer division)
n << 1                 // multiply by 2
1 << k                 // 2^k
(1 << k) - 1           // bitmask of k ones: 0...0111...1

// ===== BIT MANIPULATION PATTERNS =====
// Get bit at position k:
int bit = (n >> k) & 1;

// Set bit at position k:
n = n | (1 << k);

// Clear bit at position k:
n = n & ~(1 << k);

// Toggle bit at position k:
n = n ^ (1 << k);

// Count set bits:
int count = Integer.bitCount(n);   // built-in O(1)
// Manual:
int cnt = 0;
while (n != 0) { cnt += n & 1; n >>= 1; }
// Kernighan's: clears lowest set bit each time:
int cnt = 0;
while (n != 0) { n &= n-1; cnt++; }

// ===== XOR TRICKS =====
// XOR is: commutative (a^b=b^a), associative, a^a=0, a^0=a
a ^ b ^ a == b         // XOR same twice cancels
a ^ 0 == a             // XOR with 0 is identity

// Find single number (LC #136) — all others appear twice:
int single = 0;
for (int n : nums) single ^= n; // all pairs cancel, single remains

// Swap without temp:
a ^= b; b ^= a; a ^= b;

BFS — Breadth-First Search
Queue-based • Level-by-Level • Shortest Path (Unweighted)
Algo▼
מה זה? BFS חוקר גרף/עץ לפי רמות (Level-order). משתמש ב-Queue. למה? BFS מוצא את הנתיב הקצר ביותר בגרף לא-ממושקל (unweighted). מתי? Shortest path, Level-order traversal, Multi-source BFS (rotting oranges), Word Ladder.
מורכבות: Time O(V+E), Space O(V) לגרף; O(rows*cols) לגריד.
// ===== BFS GRAPH TEMPLATE =====
void bfs(List<List<Integer>> adj, int start, int n) {
    boolean[] visited = new boolean[n];
    Queue<Integer> queue = new ArrayDeque<>();
    queue.offer(start);
    visited[start] = true;

    while (!queue.isEmpty()) {
        int node = queue.poll();
        // process node here

        for (int nei : adj.get(node)) {
            if (!visited[nei]) {
                visited[nei] = true;
                queue.offer(nei);
            }
        }
    }
}

// ===== BFS WITH LEVELS (shortest path count) =====
int bfsShortestPath(List<List<Integer>> adj, int start, int target, int n) {
    boolean[] visited = new boolean[n];
    Queue<Integer> queue = new ArrayDeque<>();
    queue.offer(start);
    visited[start] = true;
    int steps = 0;

    while (!queue.isEmpty()) {
        int size = queue.size();    // process entire level at once
        for (int i = 0; i < size; i++) {
            int node = queue.poll();
            if (node == target) return steps;
            for (int nei : adj.get(node)) {
                if (!visited[nei]) { visited[nei] = true; queue.offer(nei); }
            }
        }
        steps++;
    }
    return -1; // not reachable
}

// ===== BFS ON GRID =====
int bfsGrid(char[][] grid, int sr, int sc, int er, int ec) {
    int rows = grid.length, cols = grid[0].length;
    int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
    boolean[][] vis = new boolean[rows][cols];
    Queue<int[]> q = new ArrayDeque<>();
    q.offer(new int[]{sr, sc});
    vis[sr][sc] = true;
    int steps = 0;
    while (!q.isEmpty()) {
        int sz = q.size();
        for (int i = 0; i < sz; i++) {
            int[] cur = q.poll();
            if (cur[0] == er && cur[1] == ec) return steps;
            for (int[] d : dirs) {
                int nr = cur[0]+d[0], nc = cur[1]+d[1];
                if (nr>=0 && nr<rows && nc>=0 && nc<cols
                    && !vis[nr][nc] && grid[nr][nc] != '#') {
                    vis[nr][nc] = true; q.offer(new int[]{nr,nc});
                }
            }
        }
        steps++;
    }
    return -1;
}

DFS — Depth-First Search
Recursive + Iterative (Stack) • Tree traversals
Algo▼
מה זה? DFS חוקר עמוק ככל האפשר לפני שחוזר (backtrack). רקורסיבי (stack הקריאות) או איטרטיבי (Stack מפורש). מתי? Connected components, cycle detection, topological sort, backtracking, tree traversals, island problems.
// ===== DFS ON GRAPH (Recursive) =====
void dfs(List<List<Integer>> adj, boolean[] visited, int node) {
    visited[node] = true;
    // process node (preorder)
    for (int nei : adj.get(node)) {
        if (!visited[nei]) dfs(adj, visited, nei);
    }
    // process node (postorder)
}

// ===== DFS ON GRID (Recursive) =====
int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
void dfsGrid(char[][] grid, boolean[][] vis, int r, int c) {
    int rows = grid.length, cols = grid[0].length;
    if (r < 0 || r >= rows || c < 0 || c >= cols) return;
    if (vis[r][c] || grid[r][c] == '0') return;
    vis[r][c] = true;
    for (int[] d : dirs) dfsGrid(grid, vis, r+d[0], c+d[1]);
}

// ===== DFS ITERATIVE (Stack) =====
void dfsIterative(List<List<Integer>> adj, int start) {
    boolean[] visited = new boolean[n];
    Deque<Integer> stack = new ArrayDeque<>();
    stack.push(start);
    while (!stack.isEmpty()) {
        int node = stack.pop();
        if (visited[node]) continue;
        visited[node] = true;
        // process node
        for (int nei : adj.get(node)) {
            if (!visited[nei]) stack.push(nei);
        }
    }
}

// ===== TREE DFS (TreeNode) — PREORDER / INORDER / POSTORDER =====
class TreeNode { int val; TreeNode left, right; }

// Preorder: root, left, right
void preorder(TreeNode node, List<Integer> res) {
    if (node == null) return;
    res.add(node.val);           // PROCESS FIRST
    preorder(node.left, res);
    preorder(node.right, res);
}

// Inorder: left, root, right (produces SORTED order for BST!)
void inorder(TreeNode node, List<Integer> res) {
    if (node == null) return;
    inorder(node.left, res);
    res.add(node.val);           // PROCESS MIDDLE
    inorder(node.right, res);
}

// Postorder: left, right, root
void postorder(TreeNode node, List<Integer> res) {
    if (node == null) return;
    postorder(node.left, res);
    postorder(node.right, res);
    res.add(node.val);           // PROCESS LAST
}

// ===== RETURN VALUE DFS (height example) =====
int height(TreeNode node) {
    if (node == null) return 0;
    int left  = height(node.left);
    int right = height(node.right);
    return 1 + Math.max(left, right);
}

Binary Search — 3 Templates
O(log n) • Sorted arrays, monotonic functions
Algo▼
מה זה? חיפוש ב-O(log n) במרחב חיפוש מסודר. בכל שלב מחצים את המרחב. 3 תבניות עיקריות: (1) חיפוש מדויק, (2) leftmost (גבול שמאלי), (3) rightmost (גבול ימני). כל תבנית עובדת על invariant שונה. תבנית 2&3 הן הכי חשובות ל-LeetCode!
// ===== TEMPLATE 1: Exact Search =====
int binarySearch(int[] arr, int target) {
    int lo = 0, hi = arr.length - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;  // NEVER: (lo+hi)/2 — can overflow!
        if (arr[mid] == target) return mid;
        if (arr[mid] <  target) lo = mid + 1;
        else                    hi = mid - 1;
    }
    return -1; // not found
}

// ===== TEMPLATE 2: Left Boundary (first position where arr[i] >= target) =====
// Returns the leftmost index where arr[idx] == target,
// OR the insertion point if not present
int leftBound(int[] arr, int target) {
    int lo = 0, hi = arr.length; // hi is exclusive!
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (arr[mid] < target) lo = mid + 1;
        else                   hi = mid;     // narrow from right
    }
    return lo; // lo == hi, insertion point
}

// ===== TEMPLATE 3: Right Boundary (last position of target) =====
int rightBound(int[] arr, int target) {
    int lo = 0, hi = arr.length;
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (arr[mid] <= target) lo = mid + 1; // narrow from left
        else                    hi = mid;
    }
    return lo - 1; // last position of target (or -1 if not found)
}

// ===== TEMPLATE: Binary Search on Answer =====
// When searching for min feasible / max feasible value:
// Define: canAchieve(x) = true if value x is feasible
// Find minimum x where canAchieve(x) is true:
int lo = minPossible, hi = maxPossible;
while (lo < hi) {
    int mid = lo + (hi - lo) / 2;
    if (canAchieve(mid)) hi = mid;     // try smaller
    else                 lo = mid + 1; // need larger
}
return lo;

// Find maximum x where canAchieve(x) is true:
while (lo < hi) {
    int mid = lo + (hi - lo + 1) / 2; // +1 to avoid infinite loop!
    if (canAchieve(mid)) lo = mid;     // try larger
    else                 hi = mid - 1;
}
return lo;
ALWAYS use mid = lo + (hi - lo) / 2 — never (lo + hi) / 2 because it can overflow. For "binary search on answer" — define a clear isValid/canAchieve function first!

// ===== KAHN'S ALGORITHM (BFS with in-degree) — most common =====
List<Integer> topoSort(int n, int[][] prerequisites) {
    List<List<Integer>> adj = new ArrayList<>();
    int[] inDegree = new int[n];
    for (int i = 0; i < n; i++) adj.add(new ArrayList<>());

    for (int[] pre : prerequisites) {
        adj.get(pre[1]).add(pre[0]);  // pre[1] -> pre[0]
        inDegree[pre[0]]++;
    }

    // Start with all nodes with in-degree 0
    Queue<Integer> queue = new ArrayDeque<>();
    for (int i = 0; i < n; i++) if (inDegree[i] == 0) queue.offer(i);

    List<Integer> order = new ArrayList<>();
    while (!queue.isEmpty()) {
        int node = queue.poll();
        order.add(node);
        for (int nei : adj.get(node)) {
            if (--inDegree[nei] == 0) queue.offer(nei);
        }
    }
    return order.size() == n ? order : new ArrayList<>(); // empty = cycle!
}

// ===== DFS POST-ORDER APPROACH =====
// 0=unvisited, 1=visiting(in stack), 2=visited
int[] state;
List<Integer> order;
boolean hasCycle = false;

boolean dfsTopoSort(List<List<Integer>> adj, int node) {
    state[node] = 1; // mark as visiting
    for (int nei : adj.get(node)) {
        if (state[nei] == 1) { hasCycle = true; return false; } // cycle!
        if (state[nei] == 0) dfsTopoSort(adj, nei);
    }
    state[node] = 2;
    order.add(node); // add in postorder = reverse topo order
    return true;
}

Dijkstra's Algorithm
Shortest Path • Weighted Graph • Non-negative weights
Algo▼
מה זה? אלגוריתם לנתיב קצר ביותר בגרף ממושקל עם משקלים אי-שליליים. משתמש ב-Min-Heap לבחור תמיד את הצומת עם המרחק הקטן ביותר. מורכבות: O((V+E) log V) עם Min-Heap. אסור: משקלים שליליים (השתמש ב-Bellman-Ford במקום).
int[] dijkstra(List<List<int[]>> adj, int src, int n) {
    // adj.get(u) = list of [neighbor, weight]
    int[] dist = new int[n];
    Arrays.fill(dist, Integer.MAX_VALUE);
    dist[src] = 0;

    // Min-heap: [distance, node]
    PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] - b[0]);
    pq.offer(new int[]{0, src});

    while (!pq.isEmpty()) {
        int[] curr = pq.poll();
        int d = curr[0], u = curr[1];

        if (d > dist[u]) continue; // outdated entry, skip

        for (int[] edge : adj.get(u)) {
            int v = edge[0], w = edge[1];
            int newDist = dist[u] + w;
            if (newDist < dist[v]) {
                dist[v] = newDist;
                pq.offer(new int[]{newDist, v});
            }
        }
    }
    return dist; // dist[i] = shortest distance from src to i
}

Algo▼
מה זה? אלגוריתם DP יעיל למציאת Subarray עם הסכום המקסימלי. Invariant: curSum = סכום ה-subarray המסתיים ב-i. אם curSum שלילי, עדיף להתחיל מחדש.
// ===== BASIC: Maximum Subarray Sum (LC #53) =====
int maxSubArray(int[] nums) {
    int curSum = nums[0], maxSum = nums[0];
    for (int i = 1; i < nums.length; i++) {
        curSum = Math.max(nums[i], curSum + nums[i]); // restart if negative
        maxSum = Math.max(maxSum, curSum);
    }
    return maxSum;
}

// ===== WITH INDICES (track start/end of best subarray) =====
int[] maxSubArrayWithIndices(int[] nums) {
    int curSum = nums[0], maxSum = nums[0];
    int start = 0, end = 0, tempStart = 0;
    for (int i = 1; i < nums.length; i++) {
        if (curSum + nums[i] < nums[i]) {
            curSum = nums[i]; tempStart = i;
        } else {
            curSum += nums[i];
        }
        if (curSum > maxSum) {
            maxSum = curSum; start = tempStart; end = i;
        }
    }
    return new int[]{maxSum, start, end};
}

// ===== CIRCULAR ARRAY variant (LC #918) =====
// Max subarray in circular array = max of:
// (1) normal max subarray (Kadane)
// (2) total sum - min subarray (Kadane reversed)
int maxSubarraySumCircular(int[] nums) {
    int totalSum = 0, maxSum = Integer.MIN_VALUE, curMax = 0;
    int minSum = Integer.MAX_VALUE, curMin = 0;
    for (int n : nums) {
        curMax = Math.max(n, curMax + n);
        maxSum = Math.max(maxSum, curMax);
        curMin = Math.min(n, curMin + n);
        minSum = Math.min(minSum, curMin);
        totalSum += n;
    }
    return maxSum > 0 ? Math.max(maxSum, totalSum - minSum) : maxSum;
}

// Math proof: when slow meets fast, move one pointer to head.
// Then both advance at speed 1 — they meet at cycle start!
ListNode detectCycle(ListNode head) {
    ListNode slow = head, fast = head;
    while (fast != null && fast.next != null) {
        slow = slow.next; fast = fast.next.next;
        if (slow == fast) {
            // Found meeting point, reset slow to head
            slow = head;
            while (slow != fast) { slow = slow.next; fast = fast.next; }
            return slow; // cycle start
        }
    }
    return null; // no cycle
}

KMP String Matching
O(n+m) time • LPS Array • Pattern Search
Algo▼
מה זה? KMP (Knuth-Morris-Pratt) מחפש pattern בתוך text ב-O(n+m) במקום O(n*m) של brute force. משתמש ב-LPS (Longest Proper Prefix which is also Suffix) array לדלג על התאמות כושלות. מתי? כשצריך חיפוש מחרוזת ב-O(n+m), also: strStr LC #28.
// ===== BUILD LPS ARRAY =====
// lps[i] = length of longest proper prefix of pattern[0..i] which is also suffix
int[] buildLPS(String pattern) {
    int m = pattern.length();
    int[] lps = new int[m];
    int len = 0, i = 1;
    while (i < m) {
        if (pattern.charAt(i) == pattern.charAt(len)) {
            lps[i++] = ++len;
        } else if (len != 0) {
            len = lps[len - 1]; // don't increment i
        } else {
            lps[i++] = 0;
        }
    }
    return lps;
}

// ===== KMP SEARCH — returns list of start indices =====
List<Integer> kmpSearch(String text, String pattern) {
    int n = text.length(), m = pattern.length();
    int[] lps = buildLPS(pattern);
    List<Integer> result = new ArrayList<>();
    int i = 0, j = 0;
    while (i < n) {
        if (text.charAt(i) == pattern.charAt(j)) {
            i++; j++;
        }
        if (j == m) {
            result.add(i - j); // found pattern at index i-j
            j = lps[j - 1];   // look for next match
        } else if (i < n && text.charAt(i) != pattern.charAt(j)) {
            if (j != 0) j = lps[j - 1];
            else        i++;
        }
    }
    return result;
}

double findMaxAverage(int[] nums, int k) {

        res.add(levelList);
        leftToRight = !leftToRight;

int orangesRotting(int[][] grid) {
    int rows = grid.length, cols = grid[0].length;
    Queue<int[]> q = new ArrayDeque<>();
    int fresh = 0;
    // Add ALL rotten oranges to queue initially
    for (int r = 0; r < rows; r++) for (int c = 0; c < cols; c++) {
        if (grid[r][c] == 2) q.offer(new int[]{r, c}); // multi-source!
        if (grid[r][c] == 1) fresh++;
    }
    if (fresh == 0) return 0;
    int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
    int minutes = 0;

        int sz = q.size(); minutes++;

                    grid[nr][nc] = 2; fresh--; q.offer(new int[]{nr,nc});
                }
            }
        }
    }
    return fresh == 0 ? minutes : -1;

// Recurrence: dp[j] = number of ways to reach sum j
int findTargetSumWays(int[] nums, int target) {

            res.add(iv);                                          // no overlap, add

int[][] insert(int[][] intervals, int[] newInterval) {
    List<int[]> res = new ArrayList<>();
    int i = 0, n = intervals.length;
    // Add all before newInterval
    while (i < n && intervals[i][1] < newInterval[0]) res.add(intervals[i++]);
    // Merge all overlapping with newInterval
    while (i < n && intervals[i][0] <= newInterval[1]) {
        newInterval[0] = Math.min(newInterval[0], intervals[i][0]);
        newInterval[1] = Math.max(newInterval[1], intervals[i][1]);
        i++;
    }
    res.add(newInterval);
    // Add all after
    while (i < n) res.add(intervals[i++]);
    return res.toArray(new int[0][]);
}

ListNode rotateRight(ListNode head, int k) {
    if (head == null) return null;
    int len = 1; ListNode tail = head;
    while (tail.next != null) { tail = tail.next; len++; } // find tail and length
    tail.next = head; // make circular
    int steps = len - k % len; // steps to new tail
    for (int i = 0; i < steps; i++) tail = tail.next;
    head = tail.next; tail.next = null; // break circle
    return head;
}

// ===== Bitmask DP — TSP-style, iterate all subsets =====
// dp[mask][i] = min cost to visit all nodes in mask, ending at i
int n = 4; // nodes
int[][] dp = new int[1 << n][n];
for (int[] row : dp) Arrays.fill(row, Integer.MAX_VALUE);
dp[1][0] = 0; // start at node 0 with node 0 visited

for (int mask = 1; mask < (1<<n); mask++) {
    for (int u = 0; u < n; u++) {
        if ((mask >> u & 1) == 0) continue; // u not in mask
        if (dp[mask][u] == Integer.MAX_VALUE) continue;
        for (int v = 0; v < n; v++) {
            if ((mask >> v & 1) == 1) continue; // v already visited
            int newMask = mask | (1 << v);
            dp[newMask][v] = Math.min(dp[newMask][v], dp[mask][u] + /* cost[u][v] */ 0);
        }
    }
}

Two Sum
Array, Hash Table • Easy
Hash Map▼
הבעיה: נתון מערך מספרים ומספר יעד, מצא שני אינדקסים שסכומם שווה ליעד.
הפתרון: עברו על המערך עם HashMap שישמור את המספר שראינו (כמפתח) והאינדקס שלו (כערך). עבור כל מספר, בדוק האם (target - current) כבר קיים ב-Map. שיטה זו נותנת פתרון ב-Pass אחד!
☕ Java Code - O(n) Time, O(n) Space
class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int complement = target - nums[i];
            if (map.containsKey(complement)) {
                return new int[] { map.get(complement), i };
            }
            map.put(nums[i], i);
        }
        return new int[]{}; // Should not reach here if guaranteed solution
    }
}

LRU Cache

Design▼
הבעיה: תכנן Data Structure שעובד לפי מדיניות Least Recently Used (הישן ביותר נמחק כשהמקום נגמר) עם פעולות ב-O(1).
הפתרון: שימוש משולב ב-HashMap (لגישה O(1)) יחד עם רשימה מקושרת דו-כיוונית (Doubly Linked List). ה-Map שומר הצבעות ישירות ל-Nodes. כל פעם כשניגשים לצומת (Put/Get) מנתקים אותו ומזיזים לראש הרשימה. כשהקיבולת מתמלאת, מוחקים מוקצה הזנב. יצירת Dummy Head & Tail מונעת טיפול ב-Edge cases מסובכים!
☕ Java Code - O(1) Time
class LRUCache {
    class Node {
        int key, val;
        Node prev, next;
        Node(int key, int val) { this.key = key; this.val = val; }
    }
    
    Node head = new Node(-1, -1);
    Node tail = new Node(-1, -1);
    int capacity;
    Map<Integer, Node> map = new HashMap<>();

    public LRUCache(int capacity) {
        this.capacity = capacity;
        head.next = tail;
        tail.prev = head;
    }
    
    // Add right after head (most recently used)
    private void addNode(Node node) {
        node.prev = head;
        node.next = head.next;
        head.next.prev = node;
        head.next = node;
    }
    
    private void removeNode(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }
    
    public int get(int key) {
        if (map.containsKey(key)) {
            Node node = map.get(key);
            removeNode(node);
            addNode(node);
            return node.val;
        }
        return -1;
    }
    
    public void put(int key, int value) {
        if (map.containsKey(key)) {
            removeNode(map.get(key));
        }
        if (map.size() == capacity) {
            map.remove(tail.prev.key);
            removeNode(tail.prev);
        }
        Node node = new Node(key, value);
        addNode(node);
        map.put(key, node);
    }
}

הבעיה: נתון מערך של k רשימות מקושרות, כל אחת ממוינת בסדר עולה. מזג (Merge) את כולן לרשימה ממוינת אחת בודדת.
הפתרון: נשתמש ב-PriorityQueue (Min-Heap). נכניס אליו את איבר הראש של כל רשימה. בכל צעד, נוציא את המינימום, נוסיף לתוצאה, ואם יש לו "המשך" (next) נאכניס את ההמשך חזרה ל-Heap. סיבוכיות הזמן תהיה O(N log k) כאשר N הוא סך כל הצמתים בכל הרשימות.
☕ Java Code - O(N log k) Time
class Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        if (lists == null || lists.length == 0) return null;
        
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (ListNode node : lists) {
            if (node != null) pq.offer(node);
        }
        
        ListNode dummy = new ListNode(0);
        ListNode current = dummy;
        
        while (!pq.isEmpty()) {
            ListNode minNode = pq.poll();
            current.next = minNode;
            current = current.next;
            
            if (minNode.next != null) {
                pq.offer(minNode.next);
            }
        }
        return dummy.next;
    }
}
